63  void MakeHistogram(TH1* h) const
64  {
65  ILOG(Info) << "Making Histogram " << h->GetName() << " out of counter" << ENDM;
66  h->Reset();
67  h->GetXaxis()->Set(Tc::size, 0, Tc::size);
68  UInt_t binx = 1;
69  for (UInt_t i = 0; i < Tc::size; i++) {
70  if (Tc::names[i].IsNull()) {
71  continue;
72  }
73  h->GetXaxis()->SetBinLabel(binx++, Tc::names[i]);
74  }
75  h->Reset();
76 #ifdef ENABLE_COUNTER_DEBUG_MODE
77  h->Print("All");
78 #endif
79  }
81  void FillHistogram(TH1* h, UInt_t biny = 0, UInt_t binz = 0) const
82  {
83  ILOG(Info) << "Filling Histogram " << h->GetName() << " out of counter" << ENDM;
84  UInt_t binx = 1;
85  for (UInt_t i = 0; i < Tc::size; i++) {
86  if (Tc::names[i].IsNull()) {
87  continue;
88  }
89 #ifdef ENABLE_COUNTER_DEBUG_MODE
90  ILOG(Info) << "Filling bin " << binx << " of position " << i << " of label " << Tc::names[i] << " with " << counter[i] << ENDM;
91 #endif
92  if (biny > 0) {
93  if (binz > 0) {
94  h->SetBinContent(binx, biny, binz, counter[i]);
95  } else {
96  h->SetBinContent(binx, biny, counter[i]);
97  }
98  } else {
99  h->SetBinContent(binx, counter[i]);
100  }
101  binx++;
102  }
103 #ifdef ENABLE_COUNTER_DEBUG_MODE
104  h->Print("All");
105 #endif
106  }
108  UInt_t Size() const { return Tc::size; }

114  uint32_t counter[Tc::size] = { 0 };
